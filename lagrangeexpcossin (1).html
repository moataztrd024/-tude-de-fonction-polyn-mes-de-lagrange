def lagrange(f,a,b,n) :
    x= np.linspace(a,b,n+1)
    x= np.poly1d ([1,0])
    
    s=0
    for i in range (n+1):
        li = 1
        for j in range (n+1):
            if (i==j) :
                continue 
            else:
                li = li*(x-x(j))/(x(i)-x(j))
        s=s+li*f(x[i])
    return s

k = [lagrange (f,0,5,n) for n in [3,5,10]]

    print (k[n])
